def regu(packet, params)
        if User.find_by_username(params[:username]) || User.find_by_email(params[:email])
                packet.data = "KO"
        else
                user = User.new(params)
                user.state = 'N'
                if user.save
                        packet.data =  "OK"
                else
packet.data = "KO"
                        log_error('[regu]  ' + user.errors.messages.to_s)
                end
        end
        packet
end
